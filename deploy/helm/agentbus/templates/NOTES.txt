----------------------------------------------------------------
Try it
----------------------------------------------------------------

Port-forward the gRPC port and run the CLI against it:

    kubectl -n {{ .Release.Namespace }} port-forward svc/{{ include "agentbus.fullname" . }} 9095:{{ .Values.service.grpcPort }}

    go install github.com/khangpt2k6/AgentBus/cmd/goqueue@latest
    goqueue publish-agent --grpc --addr localhost:9095 \
      --tenant acme --project bot --session sess-1 --agent planner \
      --type tool.call --payload '{"tool":"search"}'

    goqueue session replay --grpc --addr localhost:9095 \
      --tenant acme --project bot --session sess-1
